#include <stdio.h>

#include <limits.h>


int findLFU(int count[], int times[], int capacity) {

    int minCount = INT_MAX, minTime = INT_MAX, index = -1;

    for (int i = 0; i < capacity; i++) {

        if (count[i] < minCount || (count[i] == minCount && times[i] < minTime)) {

            minCount = count[i];

            minTime = times[i];

            index = i;

        }

    }

    return index;

}


void LFU(int pages[], int n, int capacity) {

    int frames[100], count[100], times[100], timer = 0, faults = 0;


    for (int i = 0; i < capacity; i++) {

        frames[i] = -1;

        count[i] = 0;

        times[i] = 0;

    }


    for (int i = 0; i < n; i++) {

        int pageFound = 0;

        int j;


        for (j = 0; j < capacity; j++) {

            if (frames[j] == pages[i]) {

                count[j]++;

                times[j] = timer++;

                pageFound = 1;

                break;

            }

        }


        if (!pageFound) {

            int lfu = findLFU(count, times, capacity);

            frames[lfu] = pages[i];

            count[lfu] = 1;

            times[lfu] = timer++;

            faults++;

        }


        printf("\nFrames after accessing page %d: ", pages[i]);

        for (j = 0; j < capacity; j++) {

            if (frames[j] != -1)

                printf("%d ", frames[j]);

            else

                printf("- ");

        }

    }


    printf("\n\nTotal Page Faults: %d\n", faults);

}


int main() {

    int pages[] = {7, 0, 1, 2, 0, 3, 0, 4, 2, 3, 0, 3, 2, 1, 2};

    int n = 15;

    int capacity = 3;


    LFU(pages, n, capacity);


    return 0;

}
